{

    search(A,num);
    return 0;
}
int search(int A[2][3],int y){
    int x=0;
    for(int i=0;i<2;i++){
        for(int j=0;j<3;j++){
            if(A[i][j]==y){
                printf("Element is present");
                x=1;
            }
        }
    }
    if(x==0){
        printf("Element not found");
    }
    return 0;
}

#include <stdio.h>
#include <string.h>
int main()
{
    char str[100],s[100];
    int p=0,count=0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    printf("Enter a substring: ");
    fgets(s, sizeof(s), stdin);
    for(int i=0;i<strlen(str);i++){
        if(str[i]==s[0]){
            p=i;
            count=1;
            for(int j=p;j<p+strlen(s);j++){
                if(str[j]==s[j-p]){
                    count+=1;
                }
            }
        }
        if(count==strlen(s)){
            printf("%d",p);
            break;
        }
    }
    return 0;
}

#include <stdio.h>

int main()
{
    int n,i,ele;
    int A[]={1,2,3};
    printf("Enter the element you want to search for: ");
    scanf("%d",&ele);
    search(A,ele);
    return 0;
}
int search(int A[],int y){
    int low=0;
    int n=3;
    int high=n-1,mid;
    while(low<=high){
        mid=(low+high)/2;
        if(A[mid]>y){
            high=mid-1;
        }
        else if(A[mid]<y){
            low=mid+1;
        }
        else if(A[mid]==y){
            printf("%d",mid);
            break;
        }
    }
    if(low>high){
        printf("Incorrect value");
    }
    return 0;
}

{
    int n,i;
    printf("Enter no. of elements: ");
    scanf("%d",&n);
    int A[n];
    printf("Enter elements: ");
    for(i=0;i<n;i++){
        scanf("%d",&A[i]);
    }
    int min=A[0],max=A[0];
    for(int j=0;j<n;j++){
        if(A[j]>max){
            max=A[j];
        }
        if(A[j]<min){
            min=A[j];
        }
    }
    printf("%d %d",max,min);

    return 0;
}
